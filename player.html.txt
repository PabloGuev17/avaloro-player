  <script>
    let isPlaying = false;
    const audio = document.getElementById("player");
    const icon = document.getElementById("icon");

    function toggleAudio() {
      if (!isPlaying) {
        audio.play();
        icon.src = "https://cdn-icons-png.flaticon.com/512/2089/2089610.png"; // pause icon
      } else {
        audio.pause();
        icon.src = "https://cdn-icons-png.flaticon.com/512/727/727245.png"; // play icon
      }
      isPlaying = !isPlaying;
    }
  </script>
